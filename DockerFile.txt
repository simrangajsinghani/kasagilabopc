FROM node:18

# Set the working directory in the container
WORKDIR /app

# Copy both scripts (Challenge A and Challenge B) to the container
COPY generateRandomData.js readRandomData.js package.json /app/

# Install dependencies
RUN npm install

# Generate the input file
RUN node generateRandomData.js

# Command to run Challenge B and save the output
CMD ["sh", "-c", "node readRandomData.js > /app/output.txt && tail -f /app/output.txt"]
